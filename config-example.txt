# Créez un fichier .env.local à la racine de votre projet avec ce contenu :

# Configuration de la base de données PostgreSQL
DATABASE_URL="postgresql://username:password@localhost:5432/bigmatch_db"

# Configuration Better Auth (générez une clé secrète sécurisée)
BETTER_AUTH_SECRET="your-secret-key-here-minimum-32-characters"
BETTER_AUTH_URL="http://localhost:3000"

# Configuration Google OAuth (obtenez ces valeurs depuis Google Cloud Console)
GOOGLE_CLIENT_ID="your-google-client-id"
GOOGLE_CLIENT_SECRET="your-google-client-secret"

# URL de l'application
NEXT_PUBLIC_APP_URL="http://localhost:3000"

# Instructions :
# 1. Remplacez 'username', 'password', et 'bigmatch_db' par vos vraies valeurs PostgreSQL
# 2. Générez une clé secrète sécurisée pour BETTER_AUTH_SECRET
# 3. Si vous utilisez Google Auth, configurez les variables OAuth
# 4. Enregistrez ce fichier comme .env.local à la racine du projet
